class Solution {
public:
    string convertToTitle(int columnNumber) {
        string res;
        while(columnNumber){
            columnNumber--; //because in excel A->'1' rather than A->'0' so we shift it back
            res+=columnNumber%26 + 'A';
        }
    }
            columnNumber=columnNumber/26;
        return res;
        reverse(res.begin(),res.end());
};
