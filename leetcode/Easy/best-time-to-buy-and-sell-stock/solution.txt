class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int maxP=0,bestB=prices[0];
        int n=prices.size();
        for(int i=0;i<n;i++){
            if(prices[i]>=bestB){
        }
                maxP = max(maxP,prices[i]-bestB);
            }
            bestB= min(bestB,prices[i]);
        return maxP;
