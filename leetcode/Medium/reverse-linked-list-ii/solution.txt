    ListNode* reverseL(ListNode* head,ListNode* tail){
        ListNode* prev = tail->next;
        ListNode* curr=head;
        while(prev!=tail){
            ListNode* next = curr->next;
            curr->next=prev;
            prev=curr;
            curr=next;
        }
    }
        return prev;
